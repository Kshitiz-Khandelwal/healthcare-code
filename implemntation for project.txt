## 1(a) Brute Force String Matching

```c
#include <stdio.h>
#include <string.h>

int main()
{
    char text[100], pattern[100];
    int i, j, n, m, found = 0;

    printf("Enter text: ");
    scanf("%s", text);

    printf("Enter pattern: ");
    scanf("%s", pattern);

    n = strlen(text);
    m = strlen(pattern);

    for(i = 0; i <= n - m; i++)
    {
        for(j = 0; j < m; j++)
        {
            if(text[i + j] != pattern[j])
                break;
        }

        if(j == m)
        {
            printf("Pattern found at position %d\n", i);
            found = 1;
        }
    }

    if(found == 0)
        printf("Pattern not found");

    return 0;
}

# 2. Quick Sort

```c
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

void swap(int *a, int *b)
{
    int t = *a;
    *a = *b;
    *b = t;
}

int partition(int a[], int low, int high)
{
    int pivot = a[high];
    int i = low - 1, j;

    for(j = low; j < high; j++)
    {
        if(a[j] < pivot)
        {
            i++;
            swap(&a[i], &a[j]);
        }
    }

    swap(&a[i + 1], &a[high]);

    return i + 1;
}

void quickSort(int a[], int low, int high)
{
    if(low < high)
    {
        int pi = partition(a, low, high);

        quickSort(a, low, pi - 1);
        quickSort(a, pi + 1, high);
    }
}

int main()
{
    int *a, n, i;
    clock_t start, end;
    double time_taken;

    printf("n\tTime Taken\n");

    for(n = 5000; n <= 50000; n += 5000)
    {
        a = (int*)malloc(n * sizeof(int));

        for(i = 0; i < n; i++)
            a[i] = rand() % 10000;

        start = clock();

        quickSort(a, 0, n - 1);

        end = clock();

        time_taken =
        (double)(end - start) / CLOCKS_PER_SEC;

        printf("%d\t%f\n", n, time_taken);

        free(a);
    }

    return 0;
}

# 3. Merge Sort
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

void merge(int a[], int l, int m, int r)
{
    int i = l, j = m + 1, k = 0;
    int temp[100000];

    while(i <= m && j <= r)
    {
        if(a[i] < a[j])
            temp[k++] = a[i++];
        else
            temp[k++] = a[j++];
    }

    while(i <= m)
        temp[k++] = a[i++];

    while(j <= r)
        temp[k++] = a[j++];

    for(i = l, k = 0; i <= r; i++, k++)
        a[i] = temp[k];
}

void mergeSort(int a[], int l, int r)
{
    if(l < r)
    {
        int m = (l + r) / 2;

        mergeSort(a, l, m);
        mergeSort(a, m + 1, r);

        merge(a, l, m, r);
    }
}

int main()
{
    int *a, n, i;
    clock_t start, end;
    double time_taken;

    printf("n\tTime Taken\n");

    for(n = 5000; n <= 50000; n += 5000)
    {
        a = (int*)malloc(n * sizeof(int));

        for(i = 0; i < n; i++)
            a[i] = rand() % 10000;

        start = clock();

        mergeSort(a, 0, n - 1);

        end = clock();

        time_taken = (double)(end - start) / CLOCKS_PER_SEC;

        printf("%d\t%f\n", n, time_taken);

        free(a);
    }

    return 0;
}

# 4. Kruskal’s Algorithm

```c
#include <stdio.h>

struct edge
{
    int u, v, w;
} e[10], temp;

int parent[10];

int find(int x)
{
    while(parent[x])
        x = parent[x];

    return x;
}

int main()
{
    int n, m, i, j, a, b, cost = 0;

    printf("Enter vertices and edges: ");
    scanf("%d%d", &n, &m);

    printf("Enter u v w:\n");

    for(i = 0; i < m; i++)
        scanf("%d%d%d", &e[i].u, &e[i].v, &e[i].w);

    for(i = 0; i < m - 1; i++)
    {
        for(j = 0; j < m - i - 1; j++)
        {
            if(e[j].w > e[j + 1].w)
            {
                temp = e[j];
                e[j] = e[j + 1];
                e[j + 1] = temp;
            }
        }
    }

    printf("MST Edges:\n");

    for(i = 0; i < m; i++)
    {
        a = find(e[i].u);
        b = find(e[i].v);

        if(a != b)
        {
            printf("%d-%d = %d\n", e[i].u, e[i].v, e[i].w);

            cost += e[i].w;
            parent[b] = a;
        }
    }

    printf("Minimum Cost = %d", cost);

    return 0;
}
```

# 5. Prim’s Algorithm

```c
#include <stdio.h>

int main()
{
    int cost[10][10], visited[10] = {0};
    int n, i, j, ne = 1, min, a, b;
    int mincost = 0;

    printf("Enter number of vertices: ");
    scanf("%d", &n);

    printf("Enter cost matrix:\n");

    for(i = 1; i <= n; i++)
    {
        for(j = 1; j <= n; j++)
        {
            scanf("%d", &cost[i][j]);

            if(cost[i][j] == 0)
                cost[i][j] = 999;
        }
    }

    visited[1] = 1;

    printf("MST Edges:\n");

    while(ne < n)
    {
        min = 999;

        for(i = 1; i <= n; i++)
        {
            for(j = 1; j <= n; j++)
            {
                if(cost[i][j] < min)
                {
                    if(visited[i] != 0)
                    {
                        min = cost[i][j];
                        a = i;
                        b = j;
                    }
                }
            }
        }

        if(visited[b] == 0)
        {
            printf("%d-%d = %d\n", a, b, min);

            mincost += min;
            visited[b] = 1;
            ne++;
        }

        cost[a][b] = cost[b][a] = 999;
    }

    printf("Minimum Cost = %d", mincost);

    return 0;
}

# 6. Dijkstra’s Algorithm

```c
#include <stdio.h>

int main()
{
    int cost[10][10], dist[10], visited[10];
    int n, i, j, start, min, next;

    printf("Enter number of vertices: ");
    scanf("%d", &n);

    printf("Enter cost matrix:\n");

    for(i = 0; i < n; i++)
    {
        for(j = 0; j < n; j++)
        {
            scanf("%d", &cost[i][j]);

            if(cost[i][j] == 0)
                cost[i][j] = 999;
        }
    }

    printf("Enter starting vertex: ");
    scanf("%d", &start);

    for(i = 0; i < n; i++)
    {
        dist[i] = cost[start][i];
        visited[i] = 0;
    }

    dist[start] = 0;
    visited[start] = 1;

    for(i = 1; i < n; i++)
    {
        min = 999;

        for(j = 0; j < n; j++)
        {
            if(dist[j] < min && visited[j] == 0)
            {
                min = dist[j];
                next = j;
            }
        }

        visited[next] = 1;

        for(j = 0; j < n; j++)
        {
            if(visited[j] == 0 &&
               min + cost[next][j] < dist[j])
            {
                dist[j] = min + cost[next][j];
            }
        }
    }

    printf("Shortest distances:\n");

    for(i = 0; i < n; i++)
        printf("%d -> %d = %d\n", start, i, dist[i]);

    return 0;
}

# 7. Topological Sorting

```c
#include <stdio.h>

int main()
{
    int a[10][10], indegree[10] = {0};
    int n, i, j, count = 0;

    printf("Enter number of vertices: ");
    scanf("%d", &n);

    printf("Enter adjacency matrix:\n");

    for(i = 0; i < n; i++)
    {
        for(j = 0; j < n; j++)
        {
            scanf("%d", &a[i][j]);

            if(a[i][j] == 1)
                indegree[j]++;
        }
    }

    printf("Topological Order: ");

    while(count < n)
    {
        for(i = 0; i < n; i++)
        {
            if(indegree[i] == 0)
            {
                printf("%d ", i);

                indegree[i] = -1;
                count++;

                for(j = 0; j < n; j++)
                {
                    if(a[i][j] == 1)
                        indegree[j]--;
                }
            }
        }
    }

    return 0;
}

# 8. 0/1 Knapsack

```c
#include <stdio.h>

int main()
{
    int w[10], p[10], k[10][10];
    int n, m, i, j;

    printf("Enter number of items: ");
    scanf("%d", &n);

    printf("Enter profits:\n");

    for(i = 1; i <= n; i++)
        scanf("%d", &p[i]);

    printf("Enter weights:\n");

    for(i = 1; i <= n; i++)
        scanf("%d", &w[i]);

    printf("Enter capacity: ");
    scanf("%d", &m);

    for(i = 0; i <= n; i++)
    {
        for(j = 0; j <= m; j++)
        {
            if(i == 0 || j == 0)
                k[i][j] = 0;

            else if(w[i] <= j)
            {
                if(p[i] + k[i-1][j-w[i]] > k[i-1][j])
                    k[i][j] = p[i] + k[i-1][j-w[i]];
                else
                    k[i][j] = k[i-1][j];
            }

            else
                k[i][j] = k[i-1][j];
        }
    }

    printf("Maximum Profit = %d", k[n][m]);

    return 0;
}

# 9. Subset Sum Problem

```c
#include <stdio.h>

int main()
{
    int a[10], n, sum, i, j, found = 0;

    printf("Enter number of elements: ");
    scanf("%d", &n);

    printf("Enter elements:\n");

    for(i = 0; i < n; i++)
        scanf("%d", &a[i]);

    printf("Enter required sum: ");
    scanf("%d", &sum);

    printf("Subsets are:\n");

    for(i = 0; i < (1 << n); i++)
    {
        int s = 0;

        for(j = 0; j < n; j++)
        {
            if(i & (1 << j))
                s += a[j];
        }

        if(s == sum)
        {
            found = 1;

            for(j = 0; j < n; j++)
            {
                if(i & (1 << j))
                    printf("%d ", a[j]);
            }

            printf("\n");
        }
    }

    if(found == 0)
        printf("No subset found");

    return 0;
}
```

# 10. N-Queen Problem

#include <stdlib.h>

int x[10], n;

int place(int k, int i)
{
    int j;

    for(j = 1; j < k; j++)
    {
        if(x[j] == i || abs(x[j] - i) == abs(j - k))
            return 0;
    }

    return 1;
}

void queen(int k)
{
    int i, j;

    for(i = 1; i <= n; i++)
    {
        if(place(k, i))
        {
            x[k] = i;

            if(k == n)
            {
                printf("\nSolution:\n");

                for(j = 1; j <= n; j++)
                    printf("%d ", x[j]);
            }
            else
                queen(k + 1);
        }
    }
}

    scanf("%d", &n);
